import numpy as np

def run_packing():
    n = 26
    cols = int(np.ceil(np.sqrt(n)))
    xs = (np.arange(n) % cols + 0.5) / cols
    ys = (np.arange(n) // cols + 0.5) / cols
    r0 = 0.5 / cols - 1e-3
    v0 = np.empty(3 * n)
    v0[0::3] = xs
    v0[1::3] = ys
    v0[2::3] = r0

    bounds = []
    for _ in range(n):
        bounds += [(0.0, 1.0), (0.0, 1.0), (1e-4, 0.5)]

    def neg_sum_radii(v):
        return -np.sum(v[2::3])

    cons = []
    for i in range(n):
        cons.append({"type": "ineq", "fun": lambda v, i=i: v[3*i] - v[3*i+2]})
        cons.append({"type": "ineq", "fun": lambda v, i=i: 1.0 - v[3*i] - v[3*i+2]})
        cons.append({"type": "ineq", "fun": lambda v, i=i: v[3*i+1] - v[3*i+2]})
        cons.append({"type": "ineq", "fun": lambda v, i=i: 1.0 - v[3*i+1] - v[3*i+2]})

    # Vectorized overlap constraints for performance
    def vectorized_overlap_constraint(v):
        x_centers = v[0::3]
        y_centers = v[1::3]
        r_radii = v[2::3]
        dx = x_centers[:, np.newaxis] - x_centers[np.newaxis, :]
        dy = y_centers[:, np.newaxis] - y_centers[np.newaxis, :]
        dist_sq = dx**2 + dy**2
        min_dist_sq = (r_radii[:, np.newaxis] + r_radii[np.newaxis, :])**2
        return dist_sq - min_dist_sq

    overlap_cons = []
    for i in range(n):
        for j in range(i + 1, n):
            def constraint_func(v, i=i, j=j):
                dx = v[3*i] - v[3*j]
                dy = v[3*i+1] - v[3*j+1]
                return dx*dx + dy*dy - (v[3*i+2] + v[3*j+2])**2
            overlap_cons.append({"type": "ineq", "fun": constraint_func})

    cons.extend(overlap_cons)

    # Initial optimization with perturbed guess
    def perturb_initial_guess(v):
        perturbation = 0.05 * np.random.rand(3 * n)
        return v + perturbation

    v_perturbed = perturb_initial_guess(v0)
    res = minimize(neg_sum_radii, v_perturbed, method="SLSQP", bounds=bounds,
                   constraints=cons, options={"maxiter": 500, "ftol": 1e-9})
    v = res.x if res.success else v0
    centers = np.column_stack([v[0::3], v[1::3]])
    radii = np.clip(v[2::3], 1e-6, None)

    # Dual-phase mutation strategy
    # Phase 1: Random geometric distortion
    def geometric_distortion(centers, radii):
        # Scale and shear the layout
        scale = 1.0 + 0.1 * (np.random.rand() - 0.5)
        shear = 0.1 * (np.random.rand() - 0.5)
        rotated_centers = np.zeros_like(centers)
        for i in range(n):
            x, y = centers[i]
            rotated_centers[i, 0] = x * scale + y * shear
            rotated_centers[i, 1] = y * scale
        # Ensure circles are within bounds
        for i in range(n):
            x, y = rotated_centers[i]
            r = radii[i]
            if x - r < 0:
                rotated_centers[i, 0] = r
            elif x + r > 1:
                rotated_centers[i, 0] = 1 - r
            if y - r < 0:
                rotated_centers[i, 1] = r
            elif y + r > 1:
                rotated_centers[i, 1] = 1 - r
        return rotated_centers, radii

    # Phase 2: Localized perturbation of smallest radius circles
    def shake_smallest(centers, radii):
        small_indices = np.argsort(radii)[:5]
        perturbation = 0.05 * np.random.rand(3 * n)
        perturbation[3*small_indices] = 0.1 * np.random.rand(len(small_indices)*3)

        new_v[1::3] = centers[:, 1]
        new_v[2::3] = radii
        res = minimize(neg_sum_radii, new_v, method="SLSQP", bounds=bounds,

        if res.success:
            v = res.x
            centers = np.column_stack([v[0::3], v[1::3]])
            radii = np.clip(v[2::3], 1e-6, None)
        return centers, radii

    # Apply mutation strategy
    centers, radii = geometric_distortion(centers, radii)
    centers, radii = shake_smallest(centers, radii)

    return centers, radii, float(radii.sum())
