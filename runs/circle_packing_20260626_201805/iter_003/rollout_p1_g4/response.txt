import numpy as np

def run_packing():
    n = 26
    cols = int(np.ceil(np.sqrt(n)))
    xs = (np.arange(n) % cols + 0.5) / cols
    ys = (np.arange(n) // cols + 0.5) / cols
    r0 = 0.5 / cols - 1e-3
    v0 = np.empty(3 * n)
    v0[0::3] = xs
    v0[1::3] = ys
    v0[2::3] = r0

    bounds = []
    for _ in range(n):
        bounds += [(0.0, 1.0), (0.0, 1.0), (1e-4, 0.5)]

    def neg_sum_radii(v):
        return -np.sum(v[2::3])

    cons = []
    for i in range(n):
        cons.append({"type": "ineq", "fun": lambda v, i=i: v[3*i] - v[3*i+2]})
        cons.append({"type": "ineq", "fun": lambda v, i=i: 1.0 - v[3*i] - v[3*i+2]})
        cons.append({"type": "ineq", "fun": lambda v, i=i: v[3*i+1] - v[3*i+2]})
        cons.append({"type": "ineq", "fun": lambda v, i=i: 1.0 - v[3*i+1] - v[3*i+2]})

    def vectorized_overlap_constraint(v):
        x_centers = v[0::3]
        y_centers = v[1::3]
        r_radii = v[2::3]
        dx = x_centers[:, np.newaxis] - x_centers[np.newaxis, :]
        dy = y_centers[:, np.newaxis] - y_centers[np.newaxis, :]
        dist_sq = dx**2 + dy**2
        min_dist_sq = (r_radii[:, np.newaxis] + r_radii[np.newaxis, :])**2
        return dist_sq - min_dist_sq

    overlap_cons = []
    for i in range(n):
        for j in range(i + 1, n):
            def constraint_func(v, i=i, j=j):
                dx = v[3*i] - v[3*j]
                dy = v[3*i+1] - v[3*j+1]
                return dx*dx + dy*dy - (v[3*i+2] + v[3*j+2])**2
            overlap_cons.append({"type": "ineq", "fun": constraint_func})

    cons.extend(overlap_cons)

    def perturb_initial_guess(v):
        perturbation = 0.1 * np.random.rand(3 * n)
        return v + perturbation

    def shake_smallest(v):
        r_radii = v[2::3]
        smallest_indices = np.argsort(r_radii)[:5]
        perturbation = 0.1 * np.random.rand(3 * n)
        perturbation[3*smallest_indices] = 0.2 * np.random.rand(len(smallest_indices)*3)
        return v + perturbation

    def apply_geometric_distortion(v):
        # Apply random scaling and rotation to the layout
        scale = 1.0 + 0.1 * np.random.rand()
        angle = np.pi * 0.1 * np.random.rand()
        centers = np.column_stack([v[0::3], v[1::3]])
        radii = v[2::3]
        
        # Rotate and scale the centers
        cos_theta = np.cos(angle)
        sin_theta = np.sin(angle)
        rotation_matrix = np.array([[cos_theta, -sin_theta], [sin_theta, cos_theta]])
        rotated_centers = np.dot(centers, rotation_matrix.T)
        scaled_centers = rotated_centers * scale
        
        # Adjust positions to stay within bounds
        for i in range(n):
            x, y = scaled_centers[i]
            r = radii[i]
            if x - r < 0:
                scaled_centers[i, 0] = r
            elif x + r > 1:
                scaled_centers[i, 0] = 1 - r
            if y - r < 0:
                scaled_centers[i, 1] = r
            elif y + r > 1:
                scaled_centers[i, 1] = 1 - r
        
        # Convert back to vector form
        distorted_v = np.zeros(3 * n)
        distorted_v[0::3] = scaled_centers[:, 0]
        distorted_v[1::3] = scaled_centers[:, 1]
        distorted_v[2::3] = radii
        return distorted_v

    v_initial = perturb_initial_guess(v0)
    v_perturbed = shake_smallest(v_initial)
    v_distorted = apply_geometric_distortion(v_perturbed)
    
    # First optimization
    res = minimize(neg_sum_radii, v_distorted, method="SLSQP", bounds=bounds,
                   constraints=cons, options={"maxiter": 500, "ftol": 1e-9})
    v = res.x if res.success else v_distorted
    centers = np.column_stack([v[0::3], v[1::3]])
    radii = np.clip(v[2::3], 1e-6, None)

    # Second optimization with perturbed guess
    v_perturbed_second = shake_smallest(v)
    res = minimize(neg_sum_radii, v_perturbed_second, method="SLSQP", bounds=bounds,
                   constraints=cons, options={"maxiter": 500, "ftol": 1e-9})
    v = res.x if res.success else v
    centers = np.column_stack([v[0::3], v[1::3]])
    radii = np.clip(v[2::3], 1e-6, None)

    return centers, radii, float(radii.sum())
