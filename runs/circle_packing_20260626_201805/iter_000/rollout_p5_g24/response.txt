    cols = int(np.ceil(np.sqrt(n)))

    ys = (np.arange(n) // cols + 0.5) / cols
    r0 = 0.5 / cols - 1e-3
    v0 = np.empty(3 * n)

    v0[1::3] = ys

        return -np.sum(v[2::3])
