def run_packing():
    n = 26
    cols = int(np.ceil(np.sqrt(n)))
    xs = (np.arange(n) % cols + 0.5) / cols
    ys = (np.arange(n) // cols + 0.5) / cols
    r0 = 0.5 / cols - 1e-3
    v0 = np.empty(3 * n)
    v0[0::3] = xs
    v0[1::3] = ys
    v0[2::3] = r0

    bounds = []
    for _ in range(n):
        bounds += [(0.0, 1.0), (0.0, 1.0), (1e-4, 0.5)]

    def neg_sum_radii(v):
        return -np.sum(v[2::3])

    # Add a penalty function to handle hard constraints more gracefully
    def penalty(v):
        penalty = 0.0
        # Boundary violations
        for i in range(n):
            x, y, r = v[3*i], v[3*i+1], v[3*i+2]
            if x - r < 0 or x + r > 1:
                penalty += max(0, (x - r) + 1e-12) + max(0, (x + r - 1) + 1e-12)
            if y - r < 0 or y + r > 1:
                penalty += max(0, (y - r) + 1e-12) + max(0, (y + r - 1) + 1e-12)
        # Overlap violations
        for i in range(n):
            for j in range(i + 1, n):
                dx = v[3*i] - v[3*j]
                dy = v[3*i+1] - v[3*j+1]
                dist_sq = dx*dx + dy*dy
                sum_radii = v[3*i+2] + v[3*j+2]
                if dist_sq < sum_radii*sum_radii - 1e-12:
                    penalty += max(0, dist_sq - (sum_radii*sum_radii - 1e-12))
        return penalty

    # Combine the objective with the penalty
    def total_objective(v):
        return neg_sum_radii(v) + 1000 * penalty(v)
