    cols = int(np.ceil(np.sqrt(n)))

    ys = (np.arange(n) // cols + 0.5) / cols
    r0 = 0.5 / cols - 1e-3
    v0 = np.empty(3 * n)

    v0[1::3] = ys
    v0[2::3] = r0

        return -np.sum(v[2::3])

        out_bounds = np.sum((v[0::3] - v[2::3]) < 0) + np.sum((v[0::3] + v[2::3]) > 1) + \
