    cols = int(np.ceil(np.sqrt(n)))
    xs = (np.arange(n) % cols + 0.5) / cols
    ys = (np.arange(n) // cols + 0.5) / cols
    r0 = 0.5 / cols - 1e-3
    v0 = np.empty(3 * n)

    v0[1::3] = ys
    v0[2::3] = r0

        bounds += [(0.0, 1.0), (0.0, 1.0), (1e-4, 0.5)]

        return -np.sum(v[2::3])

    def penalty_func(v):

        r = v[2::3]
        x = v[0::3]
        y = v[1::3]
        
        # Penalty for out-of-bounds circles
        for i in range(n):
            if x[i] - r[i] < 0 or x[i] + r[i] > 1:
                penalty += 1e4 * (max(0, x[i] - r[i]) + max(0, x[i] + r[i] - 1))

        for i in range(n):
            for j in range(i + 1, n):
                dx = x[i] - x[j]
                dy = y[i] - y[j]
                dist_sq = dx*dx + dy*dy
                if dist_sq < (r[i] + r[j])**2:
