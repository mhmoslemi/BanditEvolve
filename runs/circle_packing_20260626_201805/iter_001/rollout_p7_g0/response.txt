def run_packing():
    n = 26
    cols = int(np.ceil(np.sqrt(n)))
    xs = (np.arange(n) % cols + 0.5) / cols
    ys = (np.arange(n) // cols + 0.5) / cols
    r0 = 0.5 / cols - 1e-3
    v0 = np.empty(3 * n)
    v0[0::3] = xs
    v0[1::3] = ys
    v0[2::3] = r0

    for _ in range(n):
        bounds += [(0.0, 1.0), (0.0, 1.0), (1e-4, 0.5)]

        return -np.sum(v[2::3])

                r_i = v[3*i+2]
                r_j = v[3*j+2]

    radii = np.clip(v[2::3], 1e-6, None)

    # Local refinement
    def local_refinement(centers, radii):
        for _ in range(100):
            for i in range(n):

                    dx = centers[i, 0] - centers[j, 0]
                    dy = centers[i, 1] - centers[j, 1]

                    if dist < radii[i] + radii[j] - 1e-6:

                        overlap = radii[i] + radii[j] - dist
                        dx /= dist
                        dy /= dist
                        centers[i] += dx * overlap * 0.5
                        centers[j] -= dx * overlap * 0.5
                        centers[i] += dy * overlap * 0.5
                        centers[j] -= dy * overlap * 0.5

            for i in range(n):

                r = radii[i]
                if x - r < 0:

                    centers[i, 0] = 1 - r

                elif y + r > 1:
                    centers[i, 1] = 1 - r

    # Shake heuristic: perturb smallest circles and re-optimize
    small_radii_mask = radii < np.median(radii)
    small_indices = np.where(small_radii_mask)[0]

        perturbed_centers = centers.copy()
        perturbed_radii = radii.copy()

            perturbed_radii[i] = np.clip(perturbed_radii[i], 1e-6, 0.5)
