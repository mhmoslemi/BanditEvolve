import numpy as np

def run_packing():
    n = 26
    cols = int(np.ceil(np.sqrt(n)))
    xs = (np.arange(n) % cols + 0.5) / cols
    ys = (np.arange(n) // cols + 0.5) / cols
    r0 = 0.5 / cols - 1e-3
    v0 = np.empty(3 * n)
    v0[0::3] = xs
    v0[1::3] = ys
    v0[2::3] = r0

    bounds = []
    for _ in range(n):
        bounds += [(0.0, 1.0), (0.0, 1.0), (1e-4, 0.5)]

    def neg_sum_radii(v):
        return -np.sum(v[2::3])

    cons = []
    for i in range(n):
        cons.append({"type": "ineq", "fun": lambda v, i=i: v[3*i] - v[3*i+2]})
        cons.append({"type": "ineq", "fun": lambda v, i=i: 1.0 - v[3*i] - v[3*i+2]})
        cons.append({"type": "ineq", "fun": lambda v, i=i: v[3*i+1] - v[3*i+2]})
        cons.append({"type": "ineq", "fun": lambda v, i=i: 1.0 - v[3*i+1] - v[3*i+2]})

    # Vectorize the overlap constraints for better performance
    def vectorized_overlap_constraint(v):
        x_centers = v[0::3]
        y_centers = v[1::3]
        r_radii = v[2::3]
        dx = x_centers[:, np.newaxis] - x_centers[np.newaxis, :]
        dy = y_centers[:, np.newaxis] - y_centers[np.newaxis, :]
        dist_sq = dx**2 + dy**2
        min_dist_sq = (r_radii[:, np.newaxis] + r_radii[np.newaxis, :])**2
        return dist_sq - min_dist_sq

    # Convert to list of functions for each pair
    overlap_cons = []
    for i in range(n):
        for j in range(i + 1, n):
            def constraint_func(v, i=i, j=j):
                dx = v[3*i] - v[3*j]
                dy = v[3*i+1] - v[3*j+1]
                return dx*dx + dy*dy - (v[3*i+2] + v[3*j+2])**2
            overlap_cons.append({"type": "ineq", "fun": constraint_func})

    cons.extend(overlap_cons)

    # Add a radical topological reconfiguration
    def topological_reconfiguration(v):
        # Split circles into two groups: large and small
        large_radii_mask = v[2::3] > 0.15
        small_radii_mask = np.logical_not(large_radii_mask)
        
        # Generate random permutations for each group
        large_indices = np.where(large_radii_mask)[0]
        small_indices = np.where(small_radii_mask)[0]
        
        if len(large_indices) > 0 and len(small_indices) > 0:
            # Apply random permutation to each group
            np.random.shuffle(large_indices)
            np.random.shuffle(small_indices)
            
            # Reassign positions of large circles to random positions
            large_positions = np.zeros_like(v)
            large_positions[0::3][large_indices] = np.random.rand(len(large_indices))
            large_positions[1::3][large_indices] = np.random.rand(len(large_indices))
            large_positions[2::3][large_indices] = v[2::3][large_indices]
            
            # Reassign positions of small circles to random positions
            small_positions = np.zeros_like(v)
            small_positions[0::3][small_indices] = np.random.rand(len(small_indices))
            small_positions[1::3][small_indices] = np.random.rand(len(small_indices))
            small_positions[2::3][small_indices] = v[2::3][small_indices]
            
            # Combine and clip
            v_permuted = np.zeros_like(v)
            v_permuted[0::3] = np.clip(large_positions[0::3], 0.0, 1.0)
            v_permuted[1::3] = np.clip(large_positions[1::3], 0.0, 1.0)
            v_permuted[2::3] = np.clip(large_positions[2::3], 1e-4, 0.5)
            
            v_permuted[0::3][small_indices] = np.clip(small_positions[0::3][small_indices], 0.0, 1.0)
            v_permuted[1::3][small_indices] = np.clip(small_positions[1::3][small_indices], 0.0, 1.0)
            v_permuted[2::3][small_indices] = np.clip(small_positions[2::3][small_indices], 1e-4, 0.5)
            
            return v_permuted
        return v

    # Apply the topological reconfiguration to the initial guess
    v_reconfigured = topological_reconfiguration(v0)
    res = minimize(neg_sum_radii, v_reconfigured, method="SLSQP", bounds=bounds,
                   constraints=cons, options={"maxiter": 800, "ftol": 1e-9, "gtol": 1e-9})
    v = res.x if res.success else v0
    centers = np.column_stack([v[0::3], v[1::3]])
    radii = np.clip(v[2::3], 1e-6, None)
    return centers, radii, float(radii.sum())
