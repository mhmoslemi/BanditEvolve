import numpy as np

def run_packing():
    n = 26
    cols = int(np.ceil(np.sqrt(n)))
    xs = (np.arange(n) % cols + 0.5) / cols
    ys = (np.arange(n) // cols + 0.5) / cols
    r0 = 0.5 / cols - 1e-3
    v0 = np.empty(3 * n)
    v0[0::3] = xs
    v0[1::3] = ys
    v0[2::3] = r0

    bounds = []
    for _ in range(n):
        bounds += [(0.0, 1.0), (0.0, 1.0), (1e-4, 0.5)]

    def neg_sum_radii(v):
        return -np.sum(v[2::3])

    cons = []
    for i in range(n):
        cons.append({"type": "ineq", "fun": lambda v, i=i: v[3*i] - v[3*i+2]})
        cons.append({"type": "ineq", "fun": lambda v, i=i: 1.0 - v[3*i] - v[3*i+2]})
        cons.append({"type": "ineq", "fun": lambda v, i=i: v[3*i+1] - v[3*i+2]})
        cons.append({"type": "ineq", "fun": lambda v, i=i: 1.0 - v[3*i+1] - v[3*i+2]})

    # Vectorize the overlap constraints for better performance
    def vectorized_overlap_constraint(v):
        x_centers = v[0::3]
        y_centers = v[1::3]
        r_radii = v[2::3]
        dx = x_centers[:, np.newaxis] - x_centers[np.newaxis, :]
        dy = y_centers[:, np.newaxis] - y_centers[np.newaxis, :]
        dist_sq = dx**2 + dy**2
        min_dist_sq = (r_radii[:, np.newaxis] + r_radii[np.newaxis, :])**2
        return dist_sq - min_dist_sq

    # Convert to list of functions for each pair
    overlap_cons = []
    for i in range(n):
        for j in range(i + 1, n):
            def constraint_func(v, i=i, j=j):
                dx = v[3*i] - v[3*j]
                dy = v[3*i+1] - v[3*j+1]
                return dx*dx + dy*dy - (v[3*i+2] + v[3*j+2])**2
            overlap_cons.append({"type": "ineq", "fun": constraint_func})

    cons.extend(overlap_cons)

    # Apply a radical topological reconfiguration with global permutation
    def topological_reconfiguration(v):
        # Create a random permutation of circle indices
        perm = np.random.permutation(n)
        # Apply the permutation to the decision vector
        permuted_v = np.zeros_like(v)
        permuted_v[0::3] = v[0::3][perm]
        permuted_v[1::3] = v[1::3][perm]
        permuted_v[2::3] = v[2::3][perm]
        return permuted_v

    # Ensure at least one subcomponent expands radii by a fixed percentage
    def radii_expansion_constraint(v):
        # Identify a group of circles to expand
        group_size = n // 4
        group_indices = np.random.choice(n, group_size, replace=False)
        # Compute the current radii of the group
        group_radii = v[2::3][group_indices]
        # Create a target radius that is 15% larger
        target_radii = group_radii * 1.15
        # Ensure the expansion does not cause overlap
        # This is a soft constraint to guide the solver
        return np.sum((v[2::3][group_indices] - target_radii)**2)

    # Add the radii expansion constraint as an additional objective
    def radii_expansion_objective(v):
        # Identify a group of circles to expand
        group_size = n // 4
        group_indices = np.random.choice(n, group_size, replace=False)
        # Compute the current radii of the group
        group_radii = v[2::3][group_indices]
        # Create a target radius that is 15% larger
        target_radii = group_radii * 1.15
        # Return the negative of the squared difference as a penalty
        return -np.sum((v[2::3][group_indices] - target_radii)**2)

    # Create a composite objective function
    def composite_objective(v):
        # Weights: 0.9 for the main objective, 0.1 for radii expansion
        return 0.9 * neg_sum_radii(v) + 0.1 * radii_expansion_objective(v)

    # Apply the topological reconfiguration to the initial guess
    v_permuted = topological_reconfiguration(v0)
    # Run the optimization with the composite objective
    res = minimize(composite_objective, v_permuted, method="SLSQP", bounds=bounds,
                   constraints=cons, options={"maxiter": 800, "ftol": 1e-9, "gtol": 1e-9})
    v = res.x if res.success else v0
    centers = np.column_stack([v[0::3], v[1::3]])
    radii = np.clip(v[2::3], 1e-6, None)
    return centers, radii, float(radii.sum())
