    cols = int(np.ceil(np.sqrt(n)))
    xs = (np.arange(n) % cols + 0.5) / cols
    ys = (np.arange(n) // cols + 0.5) / cols
    r0 = 0.5 / cols - 1e-3
    v0 = np.empty(3 * n)

    v0[1::3] = ys
    v0[2::3] = r0

    bounds = []

        bounds += [(0.0, 1.0), (0.0, 1.0), (1e-4, 0.5)]

    def neg_sum_radii(v):
        return -np.sum(v[2::3])

    cons = []
    for i in range(n):
        cons.append({"type": "ineq", "fun": lambda v, i=i: v[3*i] - v[3*i+2]})
        cons.append({"type": "ineq", "fun": lambda v, i=i: 1.0 - v[3*i] - v[3*i+2]})
        cons.append({"type": "ineq", "fun": lambda v, i=i: v[3*i+1] - v[3*i+2]})
        cons.append({"type": "ineq", "fun": lambda v, i=i: 1.0 - v[3*i+1] - v[3*i+2]})

    def vectorized_overlap_constraint(v):
        x_centers = v[0::3]
        y_centers = v[1::3]
        r_radii = v[2::3]
        dx = x_centers[:, np.newaxis] - x_centers[np.newaxis, :]
        dy = y_centers[:, np.newaxis] - y_centers[np.newaxis, :]
        dist_sq = dx**2 + dy**2
        min_dist_sq = (r_radii[:, np.newaxis] + r_radii[np.newaxis, :])**2
        return dist_sq - min_dist_sq

    overlap_cons = []
    for i in range(n):
        for j in range(i + 1, n):
            def constraint_func(v, i=i, j=j):
                dx = v[3*i] - v[3*j]
                dy = v[3*i+1] - v[3*j+1]
                return dx*dx + dy*dy - (v[3*i+2] + v[3*j+2])**2
            overlap_cons.append({"type": "ineq", "fun": constraint_func})

    cons.extend(overlap_cons)

    def random_component_permutation(v):

        component2 = np.setdiff1d(np.arange(n), component1)
        
        # Shuffle the positions within each component
        shuffled_component1 = np.random.permutation(component1)
        shuffled_component2 = np.random.permutation(component2)
        
        # Create a new permutation of indices
        new_order = np.concatenate([shuffled_component1, shuffled_component2])
        
        # Apply the permutation to the decision vector
        permuted_v = np.copy(v)
        for i in range(n):
            permuted_v[3*i] = v[3*new_order[i]]
            permuted_v[3*i+1] = v[3*new_order[i]+1]
            permuted_v[3*i+2] = v[3*new_order[i]+2]
        
        return permuted_v

    v_shaken = v0 + 0.05 * np.random.rand(3 * n)

    # Apply component permutation heuristic
    v = random_component_permutation(v)

    # Final optimization
    res = minimize(neg_sum_radii, v, method="SLSQP", bounds=bounds,
