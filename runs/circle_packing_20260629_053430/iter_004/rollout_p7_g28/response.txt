    cols = 5
    rows = (n + cols - 1) // cols

    xs = []
    ys = []

        row = i // cols
        col = i % cols
        x = (col + 0.5) / cols
        y = (row + 0.5) / rows

            x += 0.5 / cols
        xs.append(x)
        ys.append(y)

    np.random.shuffle(ys)
