    cols = 5  # Hexagonal grid with 5 columns for 26 circles
    # Initialize centers in a hexagonal grid pattern
    xs = []
    ys = []
    for i in range(n):

        row = i // cols
        x = (col + 0.5) / cols
        y = (row + 0.5) / cols
        # Adjust for hexagonal grid (offset even rows)
        if row % 2 == 1:
            x += 0.5 / cols
        xs.append(x)
        ys.append(y)

    r0 = 0.25 / cols - 1e-3

        return -np.sum(v[2::3])
