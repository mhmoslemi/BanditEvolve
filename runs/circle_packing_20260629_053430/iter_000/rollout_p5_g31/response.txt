    cols = int(np.ceil(np.sqrt(n)))
    radius_initial = 0.02

    grid_spacing = 0.3

    for i in range(n):
        row = i // cols
        col = i % cols
        x = col * grid_spacing + 0.5 * grid_spacing
        y = row * (np.sqrt(3)/2) * grid_spacing + 0.5 * (np.sqrt(3)/2) * grid_spacing
        if (row + col) % 2 == 1:
            x += 0.5 * grid_spacing
        centers[i, 0] = x
        centers[i, 1] = y

    centers = centers / np.max(centers)

        bounds += [(0.0, 1.0), (0.0, 1.0), (1e-4, 0.5)]

        return -np.sum(v[2::3])
