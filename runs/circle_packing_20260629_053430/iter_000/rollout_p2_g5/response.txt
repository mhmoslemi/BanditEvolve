import numpy as np

def run_packing():
    n = 26
    cols = int(np.ceil(np.sqrt(n)))
    xs = (np.arange(n) % cols + 0.5) / cols
    ys = (np.arange(n) // cols + 0.5) / cols
    r0 = 0.5 / cols - 1e-3
    v0 = np.empty(3 * n)
    v0[0::3] = xs
    v0[1::3] = ys
    v0[2::3] = r0

    bounds = []
    for _ in range(n):
        bounds += [(0.0, 1.0), (0.0, 1.0), (1e-4, 0.5)]

    def objective(v):
        centers = v[0::3], v[1::3]
        radii = v[2::3]
        sum_radii = np.sum(radii)
        # Energy function: sum of radii + repulsion energy
        energy = 0
        for i in range(n):
            for j in range(i + 1, n):
                dx = centers[0][i] - centers[0][j]
                dy = centers[1][i] - centers[1][j]
                dist = np.sqrt(dx*dx + dy*dy)
                if dist < radii[i] + radii[j] - 1e-8:
                    energy += 1.0 / (radii[i] + radii[j] - dist)
        return -sum_radii + energy

    # Simulated Annealing with Basin Hopping
    from scipy.optimize import basinhopping
    def simulate_annealing():
        def annealing_func(x):
            return objective(x)
        res = basinhopping(annealing_func, v0, niter=50, T=1.0, stepsize=0.1, minimizer_kwargs={"method": "L-BFGS-B", "bounds": bounds})
        v = res.x
        return v

    v = simulate_annealing()
    centers = np.column_stack([v[0::3], v[1::3]])
    radii = np.clip(v[2::3], 1e-6, None)
    return centers, radii, float(radii.sum())
