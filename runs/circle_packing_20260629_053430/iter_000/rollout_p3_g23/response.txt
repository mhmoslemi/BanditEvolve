def run_packing():

    # Use a hexagonal grid seeding with some initial spacing
    cols = int(np.ceil(np.sqrt(n)))
    rows = int(np.ceil(n / cols))
    spacing = 0.35
    xs = []
    ys = []
    for i in range(n):
        col = i % cols
        row = i // cols
        x = (col + 0.5) * spacing
        y = (row + 0.5) * spacing
        # Adjust to fit within the unit square
        x = x / (cols * spacing)
        y = y / (rows * spacing)
        xs.append(x)
        ys.append(y)
    # Initial radii based on spacing and packing density
    r0 = 0.05
    v0 = np.empty(3 * n)
    v0[0::3] = xs
    v0[1::3] = ys

    for _ in range(n):
        bounds += [(0.0, 1.0), (0.0, 1.0), (1e-4, 0.5)]

    def neg_sum_radii(v):
        return -np.sum(v[2::3])

    for i in range(n):

        # Top boundary constraint
        cons.append({"type": "ineq", "fun": lambda v, i=i: 1.0 - v[3*i+1] - v[3*i+2]})
