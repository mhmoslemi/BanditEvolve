    cols = 5  # Fixed number of columns for hexagonal grid
    rows = 6  # Fixed number of rows for hexagonal grid
    # Generate initial positions using a hexagonal grid
    xs = np.zeros(n)
    ys = np.zeros(n)
    idx = 0
    for r in range(rows):
        for c in range(cols):
            if r % 2 == 0:
                x = c / cols
            else:
                x = (c + 0.5) / cols
            y = r / rows
            xs[idx] = x
            ys[idx] = y
            idx += 1

        return -np.sum(v[2::3])
