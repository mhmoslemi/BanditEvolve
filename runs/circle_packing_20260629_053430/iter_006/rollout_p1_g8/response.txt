    cols = 5
    rows = (n + cols - 1) // cols
    
    xs = []
    ys = []
    for i in range(n):
        row = i // cols
        col = i % cols
        x = (col + 0.5) / cols
        y = (row + 0.5) / rows
        if row % 2 == 1:
            x += 0.5 / cols
        xs.append(x)
        ys.append(y)
    
    r0 = 0.5 / cols - 1e-3

    v0[1::3] = np.array(ys)

        return -np.sum(v[2::3])

            return lambda v, i=i: v[3*i] - v[3*i+2]

            return lambda v, i=i: 1.0 - v[3*i] - v[3*i+2]

            return lambda v, i=i: v[3*i+1] - v[3*i+2]
        elif type_ == "y_max":
