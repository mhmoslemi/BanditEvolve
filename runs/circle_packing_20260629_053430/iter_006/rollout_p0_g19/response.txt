    cols = 5
    rows = (n + cols - 1) // cols
    
    xs = []
    ys = []

        col = i % cols
        x = (col + 0.5) / cols
        y = (row + 0.5) / rows
        if row % 2 == 1:
            x += 0.5 / cols
        xs.append(x)
        ys.append(y)
    
    r0 = 0.5 / cols - 1e-3
    v0 = np.empty(3 * n)

    v0[1::3] = np.array(ys)

        return -np.sum(v[2::3])

        cons.append({"type": "ineq", "fun": lambda v, i=i: 1.0 - v[3*i+1] - v[3*i+2]})

                dy = v[3*i+1] - v[3*j+1]

                min_dist_sq = (v[3*i+2] + v[3*j+2])**2

            cons.append({"type": "ineq", "fun": constraint_func})
