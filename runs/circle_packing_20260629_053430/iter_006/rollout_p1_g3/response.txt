import numpy as np

def run_packing():
    n = 26
    cols = 5
    rows = (n + cols - 1) // cols
    
    xs = []
    ys = []
    for i in range(n):
        row = i // cols
        col = i % cols
        x = (col + 0.5) / cols
        y = (row + 0.5) / rows
        if row % 2 == 1:
            x += 0.5 / cols
        xs.append(x)
        ys.append(y)
    
    r0 = 0.5 / cols - 1e-3
    v0 = np.empty(3 * n)
    v0[0::3] = np.array(xs)
    v0[1::3] = np.array(ys)
    v0[2::3] = np.full(n, r0)

    bounds = []
    for _ in range(n):
        bounds += [(0.0, 1.0), (0.0, 1.0), (1e-4, 0.5)]

    def neg_sum_radii(v):
        return -np.sum(v[2::3])

    def constraint_func(v, i, j):
        dx = v[3*i] - v[3*j]
        dy = v[3*i+1] - v[3*j+1]
        dist_sq = dx*dx + dy*dy
        min_dist_sq = (v[3*i+2] + v[3*j+2])**2
        return dist_sq - min_dist_sq

    def boundary_constraints(v, i):
        return v[3*i] - v[3*i+2]

    def boundary_constraints_top(v, i):
        return 1.0 - v[3*i] - v[3*i+2]

    def boundary_constraints_left(v, i):
        return v[3*i+1] - v[3*i+2]

    def boundary_constraints_bottom(v, i):
        return 1.0 - v[3*i+1] - v[3*i+2]

    cons = []
    for i in range(n):
        cons.append({"type": "ineq", "fun": lambda v, i=i: boundary_constraints(v, i)})
        cons.append({"type": "ineq", "fun": lambda v, i=i: boundary_constraints_top(v, i)})
        cons.append({"type": "ineq", "fun": lambda v, i=i: boundary_constraints_left(v, i)})
        cons.append({"type": "ineq", "fun": lambda v, i=i: boundary_constraints_bottom(v, i)})
    for i in range(n):
        for j in range(i + 1, n):
            cons.append({"type": "ineq", "fun": lambda v, i=i, j=j: constraint_func(v, i, j)})

    res = minimize(neg_sum_radii, v0, method="SLSQP", bounds=bounds,
                   constraints=cons, options={"maxiter": 1000, "ftol": 1e-10})
    v = res.x if res.success else v0

    # Structural decomposition: Split into two sub-components
    sub1 = v[:13]
    sub2 = v[13:]
    
    # Optimize sub-components with different constraints
    def optimize_sub(sub, bounds_sub, cons_sub):
        res = minimize(neg_sum_radii, sub, method="SLSQP", bounds=bounds_sub,
                       constraints=cons_sub, options={"maxiter": 500, "ftol": 1e-10})
        return res.x if res.success else sub
    
    # Prepare bounds and constraints for sub-components
    bounds_sub1 = [(0.0, 1.0), (0.0, 1.0), (1e-4, 0.5)] * 13
    bounds_sub2 = [(0.0, 1.0), (0.0, 1.0), (1e-4, 0.5)] * 13
    
    cons_sub1 = []
    for i in range(13):
        cons_sub1.append({"type": "ineq", "fun": lambda v, i=i: boundary_constraints(v, i)})
        cons_sub1.append({"type": "ineq", "fun": lambda v, i=i: boundary_constraints_top(v, i)})
        cons_sub1.append({"type": "ineq", "fun": lambda v, i=i: boundary_constraints_left(v, i)})
        cons_sub1.append({"type": "ineq", "fun": lambda v, i=i: boundary_constraints_bottom(v, i)})
    for i in range(13):
        for j in range(i + 1, 13):
            cons_sub1.append({"type": "ineq", "fun": lambda v, i=i, j=j: constraint_func(v, i, j)})
    
    cons_sub2 = []
    for i in range(13):
        cons_sub2.append({"type": "ineq", "fun": lambda v, i=i: boundary_constraints(v, i + 13)})
        cons_sub2.append({"type": "ineq", "fun": lambda v, i=i: boundary_constraints_top(v, i + 13)})
        cons_sub2.append({"type": "ineq", "fun": lambda v, i=i: boundary_constraints_left(v, i + 13)})
        cons_sub2.append({"type": "ineq", "fun": lambda v, i=i: boundary_constraints_bottom(v, i + 13)})
    for i in range(13):
        for j in range(i + 1, 13):
            cons_sub2.append({"type": "ineq", "fun": lambda v, i=i, j=j: constraint_func(v, i + 13, j + 13)})
    
    # Optimize sub-components
    sub1_opt = optimize_sub(sub1, bounds_sub1, cons_sub1)
    sub2_opt = optimize_sub(sub2, bounds_sub2, cons_sub2)

    v_perturbed[::3] += np.random.uniform(-perturbation, perturbation, size=26)
    v_perturbed[1::3] += np.random.uniform(-perturbation, perturbation, size=26)
    v_perturbed[2::3] += np.random.uniform(-0.01, 0.01, size=26)

        cons_perturbed.append({"type": "ineq", "fun": lambda v, i=i: boundary_constraints_left(v, i)})

    v = res_perturbed.x if res_perturbed.success else v_perturbed

    centers = np.column_stack([v[0::3], v[1::3]])
    radii = np.clip(v[2::3], 1e-6, None)
    return centers, radii, float(radii.sum())
