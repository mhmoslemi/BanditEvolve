    cols = 5

    ys = []

        x = (col + 0.5) / cols
        y = (row + 0.5) / rows
        if row % 2 == 1:
            x += 0.5 / cols
        xs.append(x)
        ys.append(y)

                min_dist_sq = (v[3*i+2] + v[3*j+2])**2
