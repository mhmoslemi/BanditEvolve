import numpy as np

def run_packing():
    n = 26
    cols = int(np.ceil(np.sqrt(n)))
    xs = (np.arange(n) % cols + 0.5) / cols
    ys = (np.arange(n) // cols + 0.5) / cols
    r0 = 0.5 / cols - 1e-3
    v0 = np.empty(3 * n)
    v0[0::3] = xs
    v0[1::3] = ys
    v0[2::3] = r0

    bounds = []
    for _ in range(n):
        bounds += [(0.0, 1.0), (0.0, 1.0), (1e-4, 0.5)]

    def neg_sum_radii(v):
        return -np.sum(v[2::3])

    cons = []
    for i in range(n):
        cons.append({"type": "ineq", "fun": lambda v, i=i: v[3*i] - v[3*i+2]})
        cons.append({"type": "ineq", "fun": lambda v, i=i: 1.0 - v[3*i] - v[3*i+2]})
        cons.append({"type": "ineq", "fun": lambda v, i=i: v[3*i+1] - v[3*i+2]})
        cons.append({"type": "ineq", "fun": lambda v, i=i: 1.0 - v[3*i+1] - v[3*i+2]})
    for i in range(n):
        for j in range(i + 1, n):
            def constraint(v, i=i, j=j):
                dx = v[3*i] - v[3*j]
                dy = v[3*i+1] - v[3*j+1]
                return dx*dx + dy*dy - (v[3*i+2] + v[3*j+2])**2
            cons.append({"type": "ineq", "fun": constraint})

    res = minimize(neg_sum_radii, v0, method="SLSQP", bounds=bounds,
                   constraints=cons, options={"maxiter": 500, "ftol": 1e-9})
    v = res.x if res.success else v0
    centers = np.column_stack([v[0::3], v[1::3]])
    radii = np.clip(v[2::3], 1e-6, None)

    # Final cleanup pass to inflate radii slightly if possible
    def cleanup(v):
        centers = np.column_stack([v[0::3], v[1::3]])
        radii = v[2::3]
        for i in range(n):
            x, y = centers[i]
            r = radii[i]
            # Try to increase radius slightly if possible
            # Check if the circle is not touching any other circle
            can_increase = True
            for j in range(n):
                if i == j:
                    continue
                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist_sq = dx*dx + dy*dy
                r_sum = r + radii[j]
                if dist_sq < r_sum*r_sum - 1e-10:
                    can_increase = False
                    break
            if can_increase:
                # Check if the circle is within the square
                if x - r < 0 or x + r > 1 or y - r < 0 or y + r > 1:
                    can_increase = False
                else:
                    # Try to increase radius by a small amount
                    new_r = r + 1e-5
                    # Check if the new radius is still valid
                    valid = True
                    for j in range(n):
                        if i == j:
                            continue
                        dx = centers[i, 0] - centers[j, 0]
                        dy = centers[i, 1] - centers[j, 1]
                        dist_sq = dx*dx + dy*dy
                        r_sum = new_r + radii[j]
                        if dist_sq < r_sum*r_sum - 1e-10:
                            valid = False
                            break
                    if valid:
                        # Check if the new radius is within the square
                        if centers[i, 0] - new_r < 0 or centers[i, 0] + new_r > 1 or \
                           centers[i, 1] - new_r < 0 or centers[i, 1] + new_r > 1:
                            valid = False
                    if valid:
                        v[3*i+2] = new_r
        return v

    v = cleanup(v)
    centers = np.column_stack([v[0::3], v[1::3]])
    radii = np.clip(v[2::3], 1e-6, None)
    return centers, radii, float(radii.sum())
