import numpy as np

def run_packing():
    n = 26
    cols = 5  # Set fixed columns for hexagonal grid (sqrt(26) ~ 5.1)
    rows = int(np.ceil(n / cols))
    
    # Generate initial positions using hexagonal grid
    xs = np.zeros(n)
    ys = np.zeros(n)
    idx = 0
    for row in range(rows):
        for col in range(cols):
            if row % 2 == 0:
                x = col / cols
            else:
                x = (col + 0.5) / cols
            y = row / rows
            xs[idx] = x
            ys[idx] = y
            idx += 1
    
    # Initial radius estimate based on spacing
    r0 = 1.0 / (cols * 2) - 1e-3
    v0 = np.empty(3 * n)
    v0[0::3] = xs
    v0[1::3] = ys
    v0[2::3] = r0

    bounds = []
    for _ in range(n):
        bounds += [(0.0, 1.0), (0.0, 1.0), (1e-4, 0.5)]

    def neg_sum_radii(v):
        return -np.sum(v[2::3])

    cons = []
    for i in range(n):
