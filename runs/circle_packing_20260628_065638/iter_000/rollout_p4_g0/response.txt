import numpy as np

def run_packing():
    n = 26
    # Hexagonal grid initialization with spiral ordering
    cols = int(np.ceil(np.sqrt(n)))
    radii_initial = 0.5 / (cols + 1)
    angles = np.linspace(0, 2 * np.pi, n, endpoint=False)
    radii = np.full(n, radii_initial)
    centers = np.zeros((n, 2))
    
    for i in range(n):
        r = radii[i]
        angle = angles[i]
        x = r * np.cos(angle)
        y = r * np.sin(angle)
        centers[i, 0] = 0.5 + x
        centers[i, 1] = 0.5 + y
    
    v0 = np.empty(3 * n)
    v0[0::3] = centers[:, 0]
    v0[1::3] = centers[:, 1]
    v0[2::3] = radii

    bounds = []
    for _ in range(n):
        bounds += [(0.0, 1.0), (0.0, 1.0), (1e-4, 0.5)]

    def neg_sum_radii(v):
        return -np.sum(v[2::3])

    cons = []
    for i in range(n):
        cons.append({"type": "ineq", "fun": lambda v, i=i: v[3*i] - v[3*i+2]})
        cons.append({"type": "ineq", "fun": lambda v, i=i: 1.0 - v[3*i] - v[3*i+2]})
        cons.append({"type": "ineq", "fun": lambda v, i=i: v[3*i+1] - v[3*i+2]})
        cons.append({"type": "ineq", "fun": lambda v, i=i: 1.0 - v[3*i+1] - v[3*i+2]})
    for i in range(n):
        for j in range(i + 1, n):
            def constraint_func(v, i=i, j=j):
                dx = v[3*i] - v[3*j]
                dy = v[3*i+1] - v[3*j+1]

    radii = np.clip(v[2::3], 1e-6, None)
