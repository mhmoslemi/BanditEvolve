    cols = int(np.ceil(np.sqrt(n)))

    angles = np.linspace(0, 2 * np.pi, n, endpoint=False)

    xs = radii * np.cos(angles)
    ys = radii * np.sin(angles)
    # Scale and shift to fit within the unit square
    xs = (xs + 0.5) / (np.max(xs) - np.min(xs)) * 0.8 + 0.1
    ys = (ys + 0.5) / (np.max(ys) - np.min(ys)) * 0.8 + 0.1
    r0 = np.min(radii) * 0.8

    v0[1::3] = ys
    v0[2::3] = r0

    bounds = []
    for _ in range(n):
        bounds += [(0.0, 1.0), (0.0, 1.0), (1e-4, 0.5)]

    def neg_sum_radii(v):
        return -np.sum(v[2::3])

    cons = []
