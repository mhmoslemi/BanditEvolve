import numpy as np

def run_packing():
    n = 26
    # Use a hexagonal grid seeding to better distribute circles
    cols = int(np.ceil(np.sqrt(n)))
    rows = int(np.ceil(n / cols))
    
    # Initialize centers in a hexagonal grid pattern
    centers = np.zeros((n, 2))
    idx = 0
    for row in range(rows):
        for col in range(cols):
            if row % 2 == 0:
                x = col / cols
            else:
                x = (col + 0.5) / cols
            y = row / rows
            centers[idx] = [x, y]
            idx += 1
    
    # Initial radius estimate based on grid spacing
    r0 = 0.25 / max(cols, rows) - 1e-3
    v0 = np.concatenate([centers.reshape(-1), np.full(n, r0)])

    for _ in range(n):
        bounds += [(0.0, 1.0), (0.0, 1.0), (1e-4, 0.5)]  # x, y, r bounds

    def neg_sum_radii(v):
        return -np.sum(v[2::3])

    cons = []
    # Add boundary constraints

        cons.append({"type": "ineq", "fun": lambda v, i=i: v[3*i] - v[3*i+2]})

        cons.append({"type": "ineq", "fun": lambda v, i=i: 1.0 - v[3*i+1] - v[3*i+2]})
