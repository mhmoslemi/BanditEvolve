import numpy as np

def run_packing():
    n = 26
    cols = int(np.ceil(np.sqrt(n)))
    # Use a hexagonal grid for initialization
    x_pos = np.linspace(0.5, 1 - 0.5, cols)
    y_pos = np.linspace(0.5, 1 - 0.5, cols)
    x_grid, y_grid = np.meshgrid(x_pos, y_pos)
    x_grid = x_grid.flatten()
    y_grid = y_grid.flatten()
    # Add a spiral pattern for better distribution
    spiral = np.zeros(n)
    for i in range(n):
        spiral[i] = 0.5 + 0.4 * np.cos(2 * np.pi * i / n)
    x_spiral = np.linspace(0.25, 0.75, n)
    y_spiral = np.linspace(0.25, 0.75, n)
    # Combine grid and spiral points
    xs = np.concatenate([x_grid, x_spiral[:n - len(x_grid)]])
    ys = np.concatenate([y_grid, y_spiral[:n - len(y_grid)]])
    # Initialize radii based on spacing between points
    r0 = np.zeros(n)
    for i in range(n):
        min_dist = np.inf
        for j in range(n):
            if i == j:
                continue
            dx = xs[i] - xs[j]
            dy = ys[i] - ys[j]
            dist = np.sqrt(dx*dx + dy*dy)
            if dist < min_dist:
                min_dist = dist
        r0[i] = min_dist / 2 - 1e-3
    v0 = np.empty(3 * n)
    v0[0::3] = xs
    v0[1::3] = ys
    v0[2::3] = r0

    bounds = []
    for _ in range(n):
        bounds += [(0.0, 1.0), (0.0, 1.0), (1e-4, 0.5)]

    def neg_sum_radii(v):
        return -np.sum(v[2::3])

    cons = []
    for i in range(n):
        cons.append({"type": "ineq", "fun": lambda v, i=i: v[3*i] - v[3*i+2]})

        cons.append({"type": "ineq", "fun": lambda v, i=i: 1.0 - v[3*i+1] - v[3*i+2]})
    for i in range(n):
        for j in range(i + 1, n):
            def constraint_func(v, i=i, j=j):
                dx = v[3*i] - v[3*j]
                dy = v[3*i+1] - v[3*j+1]

            cons.append({"type": "ineq", "fun": constraint_func})

                   constraints=cons, options={"maxiter": 1000, "ftol": 1e-9})
    v = res.x if res.success else v0

    radii = np.clip(v[2::3], 1e-6, None)
    return centers, radii, float(radii.sum())
