import numpy as np

def run_packing():
    n = 26
    cols = int(np.ceil(np.sqrt(n)))
    grid_size = 1.0 / cols
    # Hexagonal grid seeding: alternating rows with offset
    xs = np.zeros(n)
    ys = np.zeros(n)
    idx = 0
    for row in range(cols):
        for col in range(cols):
            if row % 2 == 0:
                xs[idx] = col * grid_size + grid_size / 2
            else:
                xs[idx] = (col + 0.5) * grid_size
            ys[idx] = row * grid_size + grid_size / 2
            idx += 1
    # Trim excess circles if grid has more than 26
    if idx > n:
        xs = xs[:n]
        ys = ys[:n]
    # Initial radii: based on grid spacing
    r0 = grid_size / 2 - 1e-3
    v0 = np.empty(3 * n)
    v0[0::3] = xs
    v0[1::3] = ys
    v0[2::3] = r0

    bounds = []
    for _ in range(n):
        bounds += [(0.0, 1.0), (0.0, 1.0), (1e-4, 0.5)]

    def neg_sum_radii(v):
        return -np.sum(v[2::3])

    cons = []
    for i in range(n):
        cons.append({"type": "ineq", "fun": lambda v, i=i: v[3*i] - v[3*i+2]})
        cons.append({"type": "ineq", "fun": lambda v, i=i: 1.0 - v[3*i] - v[3*i+2]})
        cons.append({"type": "ineq", "fun": lambda v, i=i: v[3*i+1] - v[3*i+2]})
        cons.append({"type": "ineq", "fun": lambda v, i=i: 1.0 - v[3*i+1] - v[3*i+2]})
    for i in range(n):
        for j in range(i + 1, n):
            def constraint_func(v, i=i, j=j):
                dx = v[3*i] - v[3*j]
                dy = v[3*i+1] - v[3*j+1]
                return dx*dx + dy*dy - (v[3*i+2] + v[3*j+2])**2
            cons.append({"type": "ineq", "fun": constraint_func})

    res = minimize(neg_sum_radii, v0, method="SLSQP", bounds=bounds,
                   constraints=cons, options={"maxiter": 1000, "ftol": 1e-9})
    v = res.x if res.success else v0

    radii = np.clip(v[2::3], 1e-6, None)
    return centers, radii, float(radii.sum())
