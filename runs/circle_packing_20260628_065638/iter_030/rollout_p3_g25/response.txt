def run_packing():

    cols = int(np.ceil(np.sqrt(n)))
    # Dynamic row/column calculation to balance aspect ratio
    # Initial layout with asymmetric spatial hashing to avoid grid-like symmetry
    rows = (n + cols - 1) // cols

    # Step 1: Initial random spatial hashing with adaptive scaling
    xs = []
    ys = []
    radii_initial = []
    for i in range(n):
        row = i // cols
        col = i % cols
        # Base grid positions with small randomized offset
        x_center = (col + 0.5) / cols
        y_center = (row + 0.5) / rows
        # Apply spatial hashing with adaptive scaling to avoid grid clustering
        spatial_hash_x = np.random.uniform(-0.12, 0.12)
        spatial_hash_y = np.random.uniform(-0.12, 0.12)
        offset = 0.05 * np.sqrt(1.0 / (row + col + 1))
        x = x_center + spatial_hash_x * offset
        y = y_center + spatial_hash_y * offset
        
        # Row stagger for non-uniform density and visual diversity
        if row % 2 == 1:
            x += 0.1 / cols * (np.random.rand() - 0.5)
        xs.append(x)
        ys.append(y)
        # Initial radii with density-aware assignment
        radius = 0.25 * (0.5 ** (row + col)) * (1 + np.random.uniform(-0.05, 0.05))
        radius = np.clip(radius, 1e-4, 0.5)
        radii_initial.append(radius)
    
    v0 = np.empty(3 * n)
    v0[0::3] = np.array(xs)
    v0[1::3] = np.array(ys)
    v0[2::3] = np.array(radii_initial)

    bounds = []
    for _ in range(n):
        bounds += [(0.0, 1.0), (0.0, 1.0), (1e-4, 0.5)]

    def neg_sum_radii(v):
        return -np.sum(v[2::3])

    # Define optimization constraints
    # Constraint type: ineq (≥ 0), representing the required inequality
    # 1. x_i - r_i ≥ 0
    # 2. 1.0 - x_i - r_i ≥ 0
    # 3. y_i - r_i ≥ 0
    # 4. 1.0 - y_i - r_i ≥ 0
    # 5. distance constraint for non-overlap: (x_i - x_j)^2 + (y_i - y_j)^2 - (r_i + r_j)^2 ≥ 0
    def build_constraints():
        cons = []
        for i in range(n):
            # Left boundary
            cons.append({"type": "ineq", "fun": lambda v, i=i: v[3*i] - v[3*i + 2]})
            # Right boundary
            cons.append({"type": "ineq", "fun": lambda v, i=i: 1.0 - v[3*i] - v[3*i + 2]})
            # Bottom boundary
            cons.append({"type": "ineq", "fun": lambda v, i=i: v[3*i+1] - v[3*i + 2]})
            # Top boundary
            cons.append({"type": "ineq", "fun": lambda v, i=i: 1.0 - v[3*i+1] - v[3*i + 2]})
            # Self-radius constraint (ensures valid radii)
            cons.append({"type": "ineq", "fun": lambda v, i=i: v[3*i+2] - 1e-5})
        for i in range(n):
            for j in range(i + 1, n):
                # Pairwise distance constraint
                cons.append({"type": "ineq",
                             "fun": lambda v, i=i, j=j:
                             (v[3*i] - v[3*j])**2 + (v[3*i+1] - v[3*j+1])**2 
                             - (v[3*i+2] + v[3*j+2])**2})
        return cons

    cons = build_constraints()
    
    # Main optimization phase with advanced tolerances
    res = minimize(neg_sum_radii, v0, method="SLSQP", bounds=bounds,
                   constraints=cons, options={
        "maxiter": 1800,
        "ftol": 1e-10,
        "gtol": 1e-10,
        "eps": 1e-6,
        "initial_step": 0.01 * np.mean(radii_initial),
        "iprint": 0,
        "disp": False
    })
    
    # Phase 2: Asymmetric spatial reconfiguration using adaptive perturbation and radius expansion with constraint-aware expansion
    if res.success:
        v = res.x
        radii_current = v[2::3]
        centers = np.column_stack([v[0::3], v[1::3]])
        
        # Step 2: Spatial Perturbation with directional gradient-awareness
        # Calculate directional gradients from pairwise distances to inform perturbation
        pairwise_grids = [ 
            np.sqrt((centers[i,0] - centers[j,0])**2 + (centers[i,1] - centers[j,1])**2) 
            for i in range(n) for j in range(i+1, n)
        ]
        dists = np.array(pairwise_grids).reshape(n, -1)
        
        # Calculate gradient vector for each point
        # This will be used for adaptive perturbation direction
        grad = np.zeros((n, 2))
        for i in range(n):
            for j in range(n):
                if i != j:
                    dx = centers[i,0] - centers[j,0]
                    dy = centers[i,1] - centers[j,1]
                    dist = np.sqrt(dx**2 + dy**2)
                    if dist > 1e-10:
                        grad[i, 0] += dx / dist * (1 / (dists[i,j] + 1e-6))  # Inverse of proximity
                        grad[i, 1] += dy / dist * (1 / (dists[i,j] + 1e-6))
        grad = grad / np.linalg.norm(grad, axis=1, keepdims=True)  # Normalize

        # Step 3: Spatial Perturbation with gradient-aware perturbation for asymmetric reconfiguration
        perturbation_strength = 0.04
        perturbation_factor = np.random.rand(n) * 0.8 + 0.2  # Vary perturbation strength for dynamic asymmetry
        perturbed_v = v.copy()
        for i in range(n):
            perturbed_v[3*i] += grad[i, 0] * perturbation_strength * perturbation_factor[i]
            perturbed_v[3*i+1] += grad[i, 1] * perturbation_strength * perturbation_factor[i]
        
        # Second optimization with perturbations
        res = minimize(neg_sum_radii, perturbed_v, method="SLSQP", bounds=bounds,
                       constraints=cons, options={
            "maxiter": 600,
            "ftol": 1e-10,
            "gtol": 1e-10,
            "eps": 1e-6,
            "iprint": 0,
            "disp": False
        })

        radii_current = v[2::3]
        centers = np.column_stack([v[0::3], v[1::3]])
        
        # Calculate pairwise distances and isolate least constrained circles
        dists = np.zeros((n, n))
        for i in range(n):

                dy = centers[i, 1] - centers[j, 1]
                dists[i, j] = np.sqrt(dx*dx + dy*dy)

                    dists[i, j] = 1e-6  # prevent division by zero
        
        # Calculate isolation metric: sum of inverse distances for each circle
        isolation = np.sum(1.0 / (dists + 1e-6), axis=1)
        isolated_indices = np.argsort(isolation)

        isolated_circle_idx = isolated_indices[-1]  # most isolated
        second_isolated_idx = isolated_indices[-2]

        current_radius_sum = np.sum(radii_current)

        target_radius_growth = 0.006 * (current_radius_sum / (n - 1))
        
        # Create a radius growth vector weighted by isolation and constraint
        radial_growth = np.zeros(n)
        growth_weight = 1.0 + (isolation - isolation.min()) / (isolation.max() - isolation.min() + 1e-6) * 0.3  # higher isolation gets more growth
        radial_growth = growth_weight * (target_radius_growth / np.sum(growth_weight))

        v_new = v.copy()

        new_radii = np.clip(new_radii, 1e-5, 0.5)

        def is_valid_config(v):

            radii = v[2::3]
            # Check boundaries
            for i in range(n):
                x, y = centers_valid[i]
                r = radii[i]
                if (x - r < -1e-12 or x + r > 1 + 1e-12
                    or y - r < -1e-12 or y + r > 1 + 1e-12):
                    return False
            # Check pairwise distances
            for i in range(n):
                for j in range(i + 1, n):
                    dx = centers_valid[i, 0] - centers_valid[j, 0]
                    dy = centers_valid[i, 1] - centers_valid[j, 1]
                    if np.sqrt(dx**2 + dy**2) < (radii[i] + radii[j]) - 1e-12:
                        return False
            return True
        
        if is_valid_config(v_new):
            v = v_new
        else:
            # If not valid, we perform gradient-based radial adjustment
            # Compute a gradient vector towards increasing sum
            v_current = v
            v_test = v_current.copy()
            v_test[2::3] = np.clip(v_new[2::3], 1e-5, 0.5)

            # Try to increase radii in a way that maintains constraint satisfaction
            # We'll use a binary search approach to find the maximum possible growth
            # Start with a step size
            eps = 0.0001
            max_grow = 0.0
            for _ in range(100):
                v_test = v_current.copy()
                v_test[2::3] = np.clip(np.array(v_new[2::3]) + max_grow * (radial_growth / radial_growth.max()), 1e-5, 0.5)
                if is_valid_config(v_test):
                    max_grow += eps
                else:
                    # backtrack
                    v_test[2::3] = np.clip(np.array(v_new[2::3]) + (max_grow - eps) * (radial_growth / radial_growth.max()), 1e-5, 0.5)
                    if not is_valid_config(v_test):
                        break
                    max_grow = max_grow - eps
            if max_grow > 0:
                v = v_current.copy()
                v[2::3] = np.clip(v_new[2::3] + np.clip(max_grow * (radial_growth / radial_growth.max()), 0, radial_growth.max()), 1e-5, 0.5)
            else:
                # If we couldn't expand, use the last valid configuration
                v = v_current.copy()
                v[2::3] = np.clip(v_new[2::3], 1e-5, 0.5)
        
        # Step 6: Final optimization with refined parameters
        res = minimize(neg_sum_radii, v, method="SLSQP", bounds=bounds,
                       constraints=cons, options={
            "maxiter": 400,
            "ftol": 1e-10,
            "gtol": 1e-10,
            "eps": 1e-6,
            "iprint": 0,
            "disp": False
        })
    
    # Final output
    v = res.x if res.success else v0
    centers = np.column_stack([v[0::3], v[1::3]])
    radii = np.clip(v[2::3], 1e-6, None)
    return centers, radii, float(radii.sum())
