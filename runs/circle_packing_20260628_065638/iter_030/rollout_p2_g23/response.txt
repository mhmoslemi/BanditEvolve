import numpy as np

def run_packing():

    cols = 5
    rows = (n + cols - 1) // cols
    
    # Initialize positions with a hybridized grid + spatially optimized random seed
    seeds = np.random.rand(n, 2)
    # Construct grid-aware seeds with spatial awareness
    xs = []
    ys = []
    for i in range(n):
        row = i // cols
        col = i % cols
        base_x = (col + 0.5) / cols
        base_y = (row + 0.5) / rows
        # Weighted perturbation: more influence on rows, less on columns
        x = base_x + seeds[i,0] * (0.12/(1 + 0.1*row)) - seeds[i,1] * (0.06/(1 + 0.1*col))
        y = base_y + seeds[i,1] * (0.12/(1 + 0.1*row)) - seeds[i,0] * (0.06/(1 + 0.1*col))
        # Alternate row staggering to optimize space
        if row % 2 == 1:
            x += 0.45 / cols * np.sign(seeds[i,0] - 0.5)
        # Prevent overlap with grid edge: apply soft constraint
        x = np.clip(x, 0.05, 0.95)
        y = np.clip(y, 0.05, 0.95)
        xs.append(x)
        ys.append(y)

    grid_width = 1.0 / cols
    grid_height = 1.0 / rows

    avg_cluster_size = grid_width * grid_height * 4

    scale_factor = 0.8 + 0.2 * (0.1 * (0.5 * cols))  # 0.8 - 0.9 based on cols
    base_radius = np.sqrt(avg_cluster_size) * scale_factor

                # Same for j
                return dist_sq[i, j] - (v[3*i + 2] + v[3*j + 2]) ** 2  # note we use the vectorized distance_sq computed earlier
            cons.append({"type": "ineq", "fun": constraint_func})
