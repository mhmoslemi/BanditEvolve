def run_packing():

    cols = 5
    rows = (n + cols - 1) // cols

    xs = []
    ys = []
    for i in range(n):
        row = i // cols
        col = i % cols
        x_center = (col + 0.5) / cols
        y_center = (row + 0.5) / rows
        x = x_center + np.random.uniform(-0.05, 0.05)
        y = y_center + np.random.uniform(-0.05, 0.05)
        if row % 2 == 1:
            x += 0.5 / cols
        xs.append(x)
        ys.append(y)

    r0 = 0.35 / cols - 1e-3
    v0 = np.empty(3 * n)
    v0[0::3] = np.array(xs)
    v0[1::3] = np.array(ys)
    v0[2::3] = np.full(n, r0)

    bounds = []

        bounds += [(0.0, 1.0), (0.0, 1.0), (1e-4, 0.5)]

        return -np.sum(v[2::3])

        for i in range(n):

        radii = v[2::3]

        min_radius_idx = np.argmin(radii)
        # Find all overlapping circles
        dists = np.zeros(n * n)
        for i in range(n):
