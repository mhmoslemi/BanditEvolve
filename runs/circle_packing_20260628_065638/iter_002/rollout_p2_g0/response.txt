def run_packing():

    cols = 5
    rows = (n + cols - 1) // cols

    xs = []
    ys = []
    for i in range(n):
        row = i // cols
        col = i % cols
        x = (col + 0.5) / cols
        y = (row + 0.5) / rows
        if row % 2 == 1:
            x += 0.5 / cols
        xs.append(x)
        ys.append(y)
    
    r0 = 0.5 / cols - 1e-3
    v0 = np.empty(3 * n)

    v0[1::3] = np.array(ys)

        return -np.sum(v[2::3])

        cons.append({"type": "ineq", "fun": lambda v, i=i: v[3*i+1] - v[3*i+2]})
        cons.append({"type": "ineq", "fun": lambda v, i=i: 1.0 - v[3*i+1] - v[3*i+2]})

                dx = v[3*i] - v[3*j]
                dy = v[3*i+1] - v[3*j+1]

                   constraints=cons, options={"maxiter": 1000, "ftol": 1e-10})
    v = res.x if res.success else v0

    radii = v[2::3]
    max_radius_index = np.argmax(radii)
    v[2::3][max_radius_index] += 0.01 * radii[max_radius_index]

    # Local optimization
    def penalty_sum_radii(v):
        radii = v[2::3]
        sum_radii = np.sum(radii)
        # Penalize out-of-bounds
        out_of_bounds = np.sum((v[0::3] - radii) < -1e-12) + np.sum((v[0::3] + radii) > 1 + 1e-12)

        overlap_penalty = 0.0
        for i in range(n):
            for j in range(i + 1, n):
                dx = v[3*i] - v[3*j]
                dy = v[3*i+1] - v[3*j+1]
