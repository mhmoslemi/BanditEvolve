    perturbed_radii = np.clip(perturbed_v[2::3], 1e-6, None)
    
    # Validate and choose the better solution
    if validate_packing(perturbed_centers, perturbed_radii)[0] and np.sum(perturbed_radii) > np.sum(radii):
        v = perturbed_v
        centers = perturbed_centers
        radii = perturbed_radii
    
    return centers, radii, float(radii.sum())
