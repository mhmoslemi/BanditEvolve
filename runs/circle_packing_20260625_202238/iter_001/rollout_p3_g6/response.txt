def run_packing():

    cols = int(np.ceil(np.sqrt(n)))
    xs = (np.arange(n) % cols + 0.5) / cols
    ys = (np.arange(n) // cols + 0.5) / cols
    r0 = 0.5 / cols - 1e-3
    v0 = np.empty(3 * n)

    v0[2::3] = r0

        return -np.sum(v[2::3])

        cons.append({"type": "ineq", "fun": lambda v, i=i: v[3*i+1] - v[3*i+2]})
        cons.append({"type": "ineq", "fun": lambda v, i=i: 1.0 - v[3*i+1] - v[3*i+2]})

                dx = v[3*i] - v[3*j]
                dy = v[3*i+1] - v[3*j+1]
                return dx*dx + dy*dy - (v[3*i+2] + v[3*j+2])**2

    # Identify constraint tightness for reordering
    constraint_tightness = []
    for i in range(n):
        for j in range(i + 1, n):
            def constraint_func_tightness(v, i=i, j=j):
                dx = v[3*i] - v[3*j]
                dy = v[3*i+1] - v[3*j+1]
                return dx*dx + dy*dy - (v[3*i+2] + v[3*j+2])**2
            # Evaluate the constraint at the initial guess to estimate tightness
            tightness = constraint_func_tightness(v0)
            constraint_tightness.append((i, j, tightness))

    # Sort constraints by tightness
    constraint_tightness.sort(key=lambda x: x[2])

    # Create a priority list of indices based on constraint tightness
    priority_indices = []
    seen = set()
    for i, j, _ in constraint_tightness:
        if i not in seen and j not in seen:
            priority_indices.append(i)
            priority_indices.append(j)
            seen.add(i)
            seen.add(j)

    # Reorder the circles based on constraint tightness
    reordered_indices = [i for i in range(n)]
    for idx in range(n):

            reordered_indices.insert(0, idx)

        reordered_v[3*i+1] = v[3*reorder_idx+1]
        reordered_v[3*i+2] = v[3*reorder_idx+2]
