    cols = int(np.ceil(np.sqrt(n)))
    xs = (np.arange(n) % cols + 0.5) / cols
    ys = (np.arange(n) // cols + 0.5) / cols
    r0 = 0.5 / cols - 1e-3
    v0 = np.empty(3 * n)
    v0[0::3] = xs
    v0[1::3] = ys
    v0[2::3] = r0

    bounds = []
    for _ in range(n):
        bounds += [(0.0, 1.0), (0.0, 1.0), (1e-4, 0.5)]

        return -np.sum(v[2::3])

        cons.append({"type": "ineq", "fun": lambda v, i=i: v[3*i+1] - v[3*i+2]})
        cons.append({"type": "ineq", "fun": lambda v, i=i: 1.0 - v[3*i+1] - v[3*i+2]})

    # Reorder constraints based on constraint tightness
    # Evaluate initial constraint tightness
    constraint_tightness = []
    for i in range(n):

        x = v0[3*i]
        y = v0[3*i+1]
        r = v0[3*i+2]
        # Tightness for x boundary
        x_tight = min(x - r, 1.0 - x - r)
        # Tightness for y boundary
        y_tight = min(y - r, 1.0 - y - r)
        # Tightness for overlap with other circles
        overlap_tight = 0.0
        for j in range(n):
            if i == j:
                continue

            dy = y - v0[3*j+1]
            dist_sq = dx*dx + dy*dy
            r_sum = r + v0[3*j+2]
            overlap_tight += max(0.0, r_sum*r_sum - dist_sq)
        constraint_tightness.append(x_tight + y_tight + overlap_tight)

    sorted_indices = np.argsort(constraint_tightness)[::-1]

    new_cons = []

                dy = v[3*i+1] - v[3*j+1]
                return dx*dx + dy*dy - (v[3*i+2] + v[3*j+2])**2

    def polish_solution(v):

        radii = np.clip(v[2::3], 1e-6, None)
        for i in range(n):
