    v0[1::3] = ys

    x_centers = v0[0::3]
    y_centers = v0[1::3]

    y_c = y_centers.reshape(n, 1)

    dy = y_c - y_c.T

    tightness = np.abs(overlap_constraints)
    tightness_indices = np.argsort(tightness, axis=None)

    reordered_indices = np.arange(n)[tightness_indices]
