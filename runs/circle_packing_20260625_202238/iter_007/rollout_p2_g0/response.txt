    ys = (np.arange(n) // cols + 0.5) / cols

    v0[1::3] = ys
    v0[2::3] = r0

    for _ in range(n):
        bounds += [(0.0, 1.0), (0.0, 1.0), (1e-4, 0.5)]

        return -np.sum(v[2::3])

        cons.append({"type": "ineq", "fun": lambda v, i=i: 1.0 - v[3*i+1] - v[3*i+2]})

                dy = v[3*i+1] - v[3*j+1]

    v_voronoi = np.empty(3 * n)

    v_voronoi[1::3] = new_points[:, 1]
    v_voronoi[2::3] = new_r0

    # Re-optimize with Voronoi-based initial guess
    res_voronoi = minimize(neg_sum_radii, v_voronoi, method="SLSQP", bounds=bounds,
                          constraints=cons, options={"maxiter": 500, "ftol": 1e-9})
    v_voronoi = res_voronoi.x if res_voronoi.success else v_voronoi

        centers = np.column_stack([v[0::3], v[1::3]])

    return centers, radii, float(radii.sum())
