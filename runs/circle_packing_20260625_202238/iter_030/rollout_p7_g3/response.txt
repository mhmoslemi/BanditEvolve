                dx = centers[i, 0] - centers[j, 0]
                dy = centers[i, 1] - centers[j, 1]
                dist = np.sqrt(dx*dx + dy*dy)
                if dist < new_radius + radii[j] - 1e-12:
                    overlap = True
                    break
            if not overlap:
                # Increase the radius and update the decision vector
                v[3*i+2] = new_radius
                # Update the radii and centers
                radii = np.clip(v[2::3], 1e-6, None)
                centers = np.column_stack([v[0::3], v[1::3]])

    # Apply non-linear context-aware spatial perturbation to the three smallest circles
    # Identify the three smallest circles
    sorted_by_radius = np.argsort(radii)
    small_circle_indices = sorted_by_radius[:3]
    
    # Make a copy of current solution to perturb
    perturbed_v = np.copy(v)
    for idx in small_circle_indices:
        i = idx
        # Scale radii of the smallest circles by a non-linear factor
        perturbed_v[3*i+2] *= 1.2
    
    # Prepare weighted objective function to prioritize small radii
    def weighted_neg_sum_radii(v):
        sum_radii = np.sum(v[2::3])
        # Weight small radii more heavily
        weighted_sum = sum_radii + 500 * np.sum(v[2::3] ** 2)
        return -weighted_sum
    
    # Re-optimize with perturbed initial guess and weighted objective
    res = minimize(weighted_neg_sum_radii, perturbed_v, method="SLSQP", bounds=bounds,
                   constraints=cons, options={"maxiter": 500, "ftol": 1e-9})
    perturbed_v = res.x if res.success else perturbed_v
    perturbed_centers = np.column_stack([perturbed_v[0::3], perturbed_v[1::3]])
    perturbed_radii = np.clip(perturbed_v[2::3], 1e-6, None)
    
    # Validate and choose the better solution
    if validate_packing(perturbed_centers, perturbed_radii)[0] and np.sum(perturbed_radii) > np.sum(radii):
        v = perturbed_v
        centers = perturbed_centers
        radii = perturbed_radii
    
    return centers, radii, float(radii.sum())
