import numpy as np
from scipy.optimize import differential_evolution

def run_packing():
    n = 26
    # Initial guess: place circles in a grid and set initial radii
    cols = int(np.ceil(np.sqrt(n)))
    xs = (np.arange(n) % cols + 0.5) / cols
    ys = (np.arange(n) // cols + 0.5) / cols
    r0 = 0.5 / cols - 1e-3
    v0 = np.empty(3 * n)
    v0[0::3] = xs
    v0[1::3] = ys
    v0[2::3] = r0

    bounds = []
    for _ in range(n):
        bounds += [(0.0, 1.0), (0.0, 1.0), (1e-4, 0.5)]

    def objective(v):
        return -np.sum(v[2::3])

    def constraints(v):
        # Bounds constraints
        for i in range(n):
            if v[3*i] - v[3*i + 2] < 0:
                return -1e-6
            if 1.0 - v[3*i] - v[3*i + 2] < 0:
                return -1e-6
            if v[3*i + 1] - v[3*i + 2] < 0:
                return -1e-6
            if 1.0 - v[3*i + 1] - v[3*i + 2] < 0:
                return -1e-6

        # Pairwise distance constraints
        for i in range(n):
            for j in range(i + 1, n):
                dx = v[3*i] - v[3*j]
                dy = v[3*i + 1] - v[3*j + 1]
                dist = np.sqrt(dx*dx + dy*dy)
                if dist < v[3*i + 2] + v[3*j + 2] - 1e-6:
                    return -1e-6
        return 0

    # Use differential evolution for better exploration
    result = differential_evolution(
        objective,
        bounds,
        constraints={"type": "ineq", "fun": constraints},
        popsize=15,
        strategy='best1bin',
        tol=1e-8,
        maxiter=300,
        mutation=(0.5, 1),
        recombination=0.8
    )

    v = result.x
    centers = np.column_stack([v[0::3], v[1::3]])
    radii = np.clip(v[2::3], 1e-6, None)
    return centers, radii, float(radii.sum())
