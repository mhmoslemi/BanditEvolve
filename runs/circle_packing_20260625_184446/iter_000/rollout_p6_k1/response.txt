import numpy as np

def run_packing():
    n = 26
    cols = int(np.ceil(np.sqrt(n)))
    # Use a hexagonal grid initialization for better spacing
    positions = []
    for i in range(n):
        row = i // cols
        col = i % cols
        x = (col + 0.5) / cols
        y = (row + 0.5) / cols + (row % 2) * (1 / (2 * cols))
        positions.append((x, y))
    # Initial radii based on spacing
    r0 = 0.5 / (cols + 1) - 1e-3
    v0 = np.empty(3 * n)
    v0[0::3] = np.array([x for x, y in positions])
    v0[1::3] = np.array([y for x, y in positions])
    v0[2::3] = np.full(n, r0)

    bounds = []
    for _ in range(n):
        bounds += [(0.0, 1.0), (0.0, 1.0), (1e-4, 0.5)]  # length 3*n

    def neg_sum_radii(v):
        return -np.sum(v[2::3])

    cons = []
    for i in range(n):

        cons.append({"type": "ineq", "fun": lambda v, i=i: v[3*i] - v[3*i+2]})
        # Right boundary constraint
        cons.append({"type": "ineq", "fun": lambda v, i=i: 1.0 - v[3*i] - v[3*i+2]})
        # Bottom boundary constraint
        cons.append({"type": "ineq", "fun": lambda v, i=i: v[3*i+1] - v[3*i+2]})
        # Top boundary constraint
        cons.append({"type": "ineq", "fun": lambda v, i=i: 1.0 - v[3*i+1] - v[3*i+2]})

                dx = v[3*i] - v[3*j]
                dy = v[3*i+1] - v[3*j+1]
